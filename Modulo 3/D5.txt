
			ESERCIZIO 1



SELECT autore.NomeAutore, 
       cantante.NomeCantante, 
       esecuzione.TitoloCanzone
FROM autore
INNER JOIN esecuzione ON esecuzione.TitoloCanzone = autore.TitoloCanzone
INNER JOIN cantante ON cantante.CodiceReg = esecuzione.CodiceReg
WHERE autore.NomeAutore=cantante.NomeCantante AND autore.NomeAutore LIKE 'D%';


SELECT DISTINCT d.TitoloAlbum, d.AnnoDisco
FROM DISCO d
LEFT JOIN ESECUZIONE e ON d.TitoloAlbum = e.TitoloCanzone
WHERE e.AnnoEsecuzione IS NULL;



SELECT autore.nome,
       autore.titolo_canz,
       cantante.nome_cantante
FROM autore
JOIN cantante on autore.nome = cantante.nome_cantante

SELECT s.MatricolaStudente, s.NomeStudente, e.CodiceCorso, d.NomeDocente
FROM STUDENTE s
JOIN ESAME e ON s.MatricolaStudente = e.MatricolaStudente
JOIN CORSO c ON e.CodiceCorso = c.CodiceCorso
JOIN DOCENTE d ON c.MatricolaDocente = d.MatricolaDocente
WHERE e.Voto > 28;



SELECT DISTINCT d.NomeDocente, c.Nome AS NomeCorso, e.SettoreScientifico
FROM DOCENTE d
JOIN CORSO c ON d.MatricolaDocente = c.MatricolaDocente
JOIN ESAME e ON e.CodiceCorso=c.CodiceCorso;
